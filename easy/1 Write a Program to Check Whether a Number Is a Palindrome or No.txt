Question: Write a Program to Check Whether a Number Is a Palindrome or No?

description: A palindrome number is a number that is equal to itself even after reversing its digits. In this program, we have to check for palindrome numbers.

my solution:
-----------

#include <iostream>
using namespace std;

bool isPalindrome(int num){
    string str1 = to_string(num);
    string str2;
    
    for(int i=str1.length()-1; i >=0; i--){
        str2 = str2 + str1[i];
    }
    
    return str1 == str2 ? true : false;
}

int main() {
    cout << isPalindrome(1221);
    return 0;
}

// Driver Code
int main()
{
	// Declaring integer

	// Declaring float
	float f_val = 30.50;
